安装使用

1. 安装依赖

npm install -g defi-automation-agent
# 或
pip install defi-automation-agent

2. 配置环境变量

export WALLET_PRIVATE_KEY="your_private_key"
export RPC_URL="https://mainnet.infura.io/v3/YOUR_KEY"
export MIN_APY=5.0  # 最低收益率阈值

3. 启动 Agent

defi-agent start --protocols aave,compound,curve
# 或使用配置文件
defi-agent start --config config.json

4. 监控运行状态

defi-agent status
defi-agent logs --tail 100

配置示例

{
  "protocols": ["aave", "compound", "curve"],
  "strategies": {
    "rebalance": {
      "enabled": true,
      "threshold": 0.5,
      "interval": 3600
    },
    "arbitrage": {
      "enabled": true,
      "minProfit": 0.01
    }
  },
  "risk": {
    "maxSlippage": 0.005,
    "maxGasPrice": 100,
    "emergencyStop": true
  }
}
